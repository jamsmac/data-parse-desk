  if (!statusColumn) {

      {
        id: 'todo',
        title: 'To Do',
        cards: rows.slice(0, Math.ceil(rows.length / 3)).map(...)
      },
      {
        id: 'in-progress',

  const statuses = [...new Set(rows.map(r => r[statusColumn.column_name]))].filter(Boolean);

    id: status,
    title: status,
